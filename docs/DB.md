## 테이블 생성 쿼리

```sql
CREATE TABLE users (
    user_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_name VARCHAR2(50) NOT NULL,
    login_id VARCHAR2(50) NOT NULL UNIQUE
    password VARCHAR2(255) NOT NULL,
    nickname VARCHAR2(50) NOT NULL,
    role VARCHAR2(20) NOT NULL,
    phone VARCHAR2(30)
);

CREATE TABLE admins (
    admin_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id NUMBER NOT NULL,

    CONSTRAINT fk_admins_users
        FOREIGN KEY (user_id)
        REFERENCES users(user_id)
        ON DELETE CASCADE,

    CONSTRAINT uk_admins_user_id
        UNIQUE (user_id)
);

CREATE TABLE boards (
    board_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    admin_id NUMBER NOT NULL,

    CONSTRAINT fk_boards_admins
        FOREIGN KEY (admin_id)
        REFERENCES admins(admin_id)
        ON DELETE CASCADE
);

CREATE TABLE posts (
    post_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id NUMBER NOT NULL,
    board_id NUMBER NOT NULL,
    title VARCHAR2(200) NOT NULL,
    content_posts CLOB NOT NULL,
    price NUMBER NOT NULL,
    category VARCHAR2(50),
    location VARCHAR2(100),
    status VARCHAR2(20) NOT NULL,
    created_at_posts TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,

    CONSTRAINT fk_posts_users
        FOREIGN KEY (user_id)
        REFERENCES users(user_id)
        ON DELETE CASCADE,

    CONSTRAINT fk_posts_boards
        FOREIGN KEY (board_id)
        REFERENCES boards(board_id)
        ON DELETE CASCADE
);

CREATE TABLE post_img (
    image_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    post_id NUMBER NOT NULL,
    original_name VARCHAR2(255),
    stored_name VARCHAR2(255) NOT NULL,
    image_url VARCHAR2(1000) NOT NULL,
    created_at_img TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,

    CONSTRAINT fk_post_img_posts
        FOREIGN KEY (post_id)
        REFERENCES posts(post_id)
        ON DELETE CASCADE
);

CREATE TABLE comments (
    comments_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    user_id NUMBER NOT NULL,
    post_id NUMBER NOT NULL,
    content_comments CLOB NOT NULL,
    created_at_comments TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at_comments TIMESTAMP,

    CONSTRAINT fk_comments_users
        FOREIGN KEY (user_id)
        REFERENCES users(user_id)
        ON DELETE CASCADE,

    CONSTRAINT fk_comments_posts
        FOREIGN KEY (post_id)
        REFERENCES posts(post_id)
        ON DELETE CASCADE
);
```
